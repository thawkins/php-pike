  97.                 $classPrefix = ucfirst($moduleName) . '_';

  99.             // Iterate over controllers found in the controller directory

  101.             foreach ($directoryIterator as $file) {

  103.                 if ($file->isDot() || substr($file->getFilename(), 0, 1) == '.' || $file->isDir()) {

  106.  
  107.                 // Initialize the reflection of the controller class
  108.                 require_once $file->getPathname();

  112.

  114.                 $methods = $classReflection->getMethods(ReflectionMethod::IS_PUBLIC);
  115.                 foreach ($methods as $method) {
  116.                     $resourceAttributes = array();
  117.  
  118.                     // Ignore methods that don't have an "Action" suffix
  119.                     if ($method->class == $className && substr($method->name, -6) == 'Action') {
  120.                         $resourceAttributes = $this->_getResourceAttributesFromActionMethod($method);
  121.                         $actionName = $this->_getInflectedActionName(substr($method->name, 0, -6));
  122.                         $resources[$moduleName][strtolower($controllerName)][$actionName]
  123.                                 = $resourceAttributes;

  127.         }
  128.  
  129.         $resources = $this->_excludeErrorHandlerResource($resources);
  130.         return $resources;
  131.     }

  187.         if ($docComment) {

  197.                         break;
  198.                     case 'roles':

  202.                                 $roles = explode('|', $tagValue);
  203.                                 foreach ($roles as &$role) {
  204.                                     $role = strtolower(trim(str_replace(' ', '', $role)));
  205.                                 }
  206.                                 $resourceAttributes[$attribute] = $roles;
  207.                             }
  208.                         }
  209.                         break;
  210.                     default:
  211.                         if ($actionMethod->getDocblock()->hasTag($attribute)) {
  212.                             $tagValue = $actionMethod->getDocblock()->getTag($attribute)->getDescription();
  213.                             if (trim($tagValue) != '') {
  214.                                 $resourceAttributes[$attribute] = $tagValue;
  215.                             }
  216.                         }
  217.                 }
  218.             }
  219.         }

  221.         return $resourceAttributes;
