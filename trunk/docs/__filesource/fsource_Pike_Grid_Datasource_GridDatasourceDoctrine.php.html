  113.     /**
  114.      *
  115.      * Looks up in the AST what select expression we use and analyses which
  116.      * fields are used. This is passed thru the datagrid for displaying fieldnames.
  117.      *
  118.      *
  119.      * @return array 
  120.      */
  121.     private function _setColumns()
  122.     {
  123.         $this->columns = new Pike_Grid_Datasource_Columns();
  124.  
  125.         $selectClause = $this->_query->getAST()->selectClause;
  126.         if(count($selectClause->selectExpressions) == 0) {
  127.             throw new Pike_Exception('The grid query should contain at least one column, none found.');
  128.         }
  129.  
  130.         /* @var $selExpr Doctrine\ORM\Query\AST\SelectExpression */
  131.         foreach($selectClause->selectExpressions as $selExpr) {
  132.  
  133.             /**
  134.              * Some magic needs to happen here. If $selExpr isn't a instanceof
  135.              * Doctrine\ORM\Query\AST\PathExpression then it might be a custom
  136.              * expression like a vendor specific function. We could use the 
  137.              * $selExpr->fieldIdentificationVariable member which is the alias
  138.              * given to the special expression but sorting at this field
  139.              * could cause strange effects.
  140.              * 
  141.              * For instance: SELECT DATE_FORMAT(datefield,"%Y") AS alias FROM sometable
  142.              * 
  143.              * When you say: ORDER BY alias DESC
  144.              * 
  145.              * You could get other results when you do:
  146.              * 
  147.              * ORDER BY sometable.datefield DESC
  148.              * 
  149.              * My idea is to rely on the alias field by default because it would 
  150.              * suite for most queries. If someone would like to retrieve this expression
  151.              * specific info they can add a field filter. $ds->addFieldFilter(new DateFormat_Field_Filter(), 'expression classname');
  152.              * 
  153.              * And info of the field would be extracted from this class, something like that.
  154.              */
  155.             
  156.             $expr = $selExpr->expression;
  157.             
  158.             /* @var $expr Doctrine\ORM\Query\AST\PathExpression */
  159.             if($expr instanceof Doctrine\ORM\Query\AST\PathExpression) {
  160.                 $alias = $expr->identificationVariable;
  161.                 $name = ($selExpr->fieldIdentificationVariable === null) ? $expr->field : $selExpr->fieldIdentificationVariable;
  162.                 $label = ($selExpr->fieldIdentificationVariable === null) ? $name : $selExpr->fieldIdentificationVariable;
  163.                 $index = (strlen($alias) > 0 ? ($alias . '.') : '') . $expr->field;
  164.             } else {
  165.                 $name = $selExpr->fieldIdentificationVariable;
  166.                 $label = $name;
  167.                 $index = null;
  168.             }
  169.             
  170.             $this->columns->add($name, $label, $index);
  171.         }
  172.         
  173.     }

  209.     /**
  210.      *
  211.      * Look if there is default sorting defined in the original query by asking the AST. Defining

  216.     public function getDefaultSorting()
  217.     {
  218.         if(null !== $this->_query->getAST()->orderByClause) {
  219.            //support for 1 field only
  220.             $orderByClause = $this->_query->getAST()->orderByClause;
  221.  
  222.             /* @var $orderByItem Doctrine\ORM\Query\AST\OrderByItem */
  223.             $orderByItem = $orderByClause->orderByItems[0];
  224.  
  225.             if($orderByItem->expression instanceof \Doctrine\ORM\Query\AST\PathExpression) {
  226.                 $alias = $orderByItem->expression->identificationVariable;
  227.                 $field = $orderByItem->expression->field;
  228.  
  229.                 $data['index'] = (strlen($alias) > 0 ? $alias . '.' : '') . $field;
  230.                 $data['direction'] = $orderByItem->type;
  231.  
  232.                 return $data;
  233.             }
  234.         }
  235.  
  236.         return null;
  237.     }

  239.     /**
  240.      *
  241.      * Renders a single row and adds it to the internal data array
  242.      *
  243.      * @param array $row 
  244.      * @return void 
  245.      */
  246.     private function _renderRow($row) {
  247.         foreach($this->columns as $index=>$column) {
  248.  
  249.             if(array_key_exists('data', $column)) {
  250.                 if(is_callable($column['data'])) {
  251.                     $row[$index] = $column['data']($row);
  252.                 } else {
  253.                     array_walk($row, function($value, $key) use (&$column) {
  254.                         $column['data'] = str_replace('{' . strtolower($key) . '}', $value, $column['data']);
  255.                     });
  256.  
  257.                     $row[$index] = $column['data'];
  258.                 }
  259.             } elseif(array_key_exists($index, $row)) {
  260.                 continue;
  261.             } else {
  262.                 throw new Pike_Exception('Cannot render data for column '.$index);
  263.             }
  264.         }
  265.  
  266.         /**
  267.          * Sort the row data specified by the column positions
  268.          */
  269.         $columns = $this->columns;
  270.  
  271.         uksort($row, function($a, $b) use ($columns) {
  272.             if($columns[$a]['position'] > $columns[$b]['position']) {
  273.                 return 1;
  274.             } elseif($columns[$a]['position'] < $columns[$b]['position']) {
  275.                 return -1;
  276.             } else {
  277.                 return 0;
  278.             }
  279.         });
  280.  
  281.         $record = array();
  282.         $record['cell'] = array_values($row);
  283.  
  284.         if(null !== $this->_identifierColumn) {                
  285.             $record['id'] = $this->_identifierColumn['data']($row);
  286.         }
  287.  
  288.         $this->_data['rows'][] = $record;
  289.     }

  291.     /**
  292.      *
  293.      * Get the initialized query hints based on the user-interface jqgrid parameters.
  294.      *
  295.      * @return array Doctrine compatible hints array
  296.      */
  297.     private function _getQueryHints() {
  298.         $hints = array();
  299.  
  300.         /**
  301.          * Sorting if defined
  302.          */
  303.         if(array_key_exists('sidx', $this->_params) && strlen($this->_params['sidx']) > 0) {
  304.             $onSort = $this->onOrder;
  305.             $hints = array_merge_recursive($hints, $onSort($this->_params, $this));
  306.         }
  307.  
  308.         if(array_key_exists('filters', $this->_params) && (array_key_exists('_search', $this->_params) && $this->_params['_search'] == true)) {
  309.             $onFilter = $this->onFilter;
  310.             $hints = array_merge_recursive($hints, $onFilter($this->_params, $this));
  311.         }
  312.  
  313.         return $hints;
  314.     }

  316.     /**
  317.      *
  318.      * Returns a JSON string useable for JQuery Grid. This grids interprets this
  319.      * data and is able to draw a grid.
  320.      *
  321.      * @return string JSON data
  322.      */
  323.     public function getJSON()
  324.     {
  325.         $offset = $this->_limitPerPage * ($this->_params['page'] - 1);
  326.         $hints = $this->_getQueryHints();
  327.  
  328.         $count = Pike_Grid_Datasource_Doctrine_Paginate::getTotalQueryResults($this->_query, $hints);
  329.         
  330.         $paginateQuery = Pike_Grid_Datasource_Doctrine_Paginate::getPaginateQuery(
  331.                 $this->_query,
  332.                 $offset,
  333.                 $this->_limitPerPage,
  334.                 $hints
  335.         );
  336.  
  337.         $result = $paginateQuery->getResult();
  338.  
  339.         $this->_data = array();
  340.         $this->_data['page'] = (int)$this->_params['page'];
  341.         $this->_data['total'] = ceil($count / $this->_limitPerPage);
  342.         $this->_data['records'] = $count;
  343.         $this->_data['rows'] = array();
  344.  
  345.         foreach($result as $row) {            
  346.             $this->_renderRow($row);           
  347.         }
  348.  
  349.         return json_encode($this->_data);
  350.     }
